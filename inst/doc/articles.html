All Features

This vignette walks through the advanced features of panelforest using a single clinical trial subgroup dataset. Each section introduces one feature and shows working code that produces a rendered plot.

Data

df <- data.frame(
  subgroup = c("Overall", "Age < 65", "Age >= 65",
               "Male", "Female", "ECOG 0", "ECOG 1-2"),
  n        = c(312L, 163L, 149L, 180L, 132L, 108L, 204L),
  events   = c(187L, 91L, 96L, 104L, 83L, 58L, 129L),
  HR       = c(0.68, 0.61, 0.76, 0.64, 0.74, 0.53, 0.76),
  LCI      = c(0.54, 0.44, 0.57, 0.47, 0.53, 0.34, 0.59),
  UCI      = c(0.85, 0.83, 1.01, 0.86, 1.04, 0.82, 0.98),
  pval     = c(0.001, 0.003, 0.062, 0.008, 0.098, 0.044, 0.012),
  resp_rate = c(0.42, 0.46, 0.38, 0.44, 0.40, 0.51, 0.38),
  mean_diff = c(-0.32, -0.41, -0.22, -0.36, -0.27, -0.48, -0.25),
  lci_diff  = c(-0.55, -0.68, -0.46, -0.60, -0.52, -0.75, -0.48),
  uci_diff  = c(-0.09,  -0.14, 0.02, -0.12, 0.02, -0.21, -0.02),
  stringsAsFactors = FALSE
)

1 Log-scale CI panel with axis

When plotting hazard ratios the x-axis should be on a log scale. Set trans = "log", supply explicit xlim and breaks, enable the axis with show_axis = TRUE, and label it with axis_label. The reference line at 1 (no effect) is drawn automatically via ref_line = 1.

p1 <- forest_plot(df) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>
  add_text_ci("HR", "LCI", "UCI", header = "HR (95% CI)", width = 2) |>
  add_ci("HR", "LCI", "UCI",
    trans      = "log",
    xlim       = c(0.2, 3),
    breaks     = c(0.25, 0.5, 1, 2),
    show_axis  = TRUE,
    axis_label = "Hazard Ratio (log scale)",
    ref_line   = 1,
    header     = "Forest Plot"
  )

size <- fp_size(p1)
fp_render(p1)

Key parameters:

Parameter Role
trans = "log" Log10 scale for the x-axis
xlim Sets both the visible axis range and the truncation boundary
breaks Explicit tick positions (must be positive)
show_axis = TRUE Renders the x-axis below the CI panel
axis_label Text printed under the axis
ref_line Vertical reference line (must be > 0 on log scale)

2 Truncation arrows

When a confidence interval whisker extends beyond xlim panelforest draws an arrow rather than an impossible whisker. By default the truncation boundary equals xlim. Use the separate truncate argument when you want the axis to show a wider range but still clip CIs to a narrower window.

Arrow appearance is controlled by three parameters:

  • arrow_length — arrowhead size in inches (default 0.08)
  • arrow_type — "open" or "closed" (default "closed")
  • arrow_angle — half-angle of the arrowhead in degrees (default 30)
# Age >= 65 and Female have UCI > 1 — their upper whiskers will be clipped
# at 0.95 while the axis still shows up to 1.2.
p2 <- forest_plot(df) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>
  add_ci("HR", "LCI", "UCI",
    trans        = "log",
    xlim         = c(0.2, 1.2),
    truncate     = c(0.2, 0.95),
    breaks       = c(0.25, 0.5, 1),
    show_axis    = TRUE,
    ref_line     = 1,
    header       = "HR (truncated at 0.95)",
    arrow_length = 0.1,
    arrow_type   = "open",
    arrow_angle  = 25
  )

fp_render(p2)

xlim = c(0.2, 1.2) defines the full axis extent. truncate = c(0.2, 0.95) clips the CI whiskers more aggressively — any upper bound above 0.95 is replaced by an outward-pointing arrow.

3 Favors annotations

favors_left and favors_right add directional labels and arrows below the CI panel. They are typically paired with show_axis = TRUE. favors_span (0–1) controls how far the arrows extend from the reference line, and favors_gap adds space between the arrow origin and the reference line.

p3 <- forest_plot(df) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>
  add_ci("HR", "LCI", "UCI",
    trans         = "log",
    xlim          = c(0.2, 3),
    breaks        = c(0.25, 0.5, 1, 2),
    ref_line      = 1,
    show_axis     = TRUE,
    axis_label    = "Hazard Ratio",
    favors_left   = "Favors Treatment",
    favors_right  = "Favors Control",
    favors_span   = 0.65,
    favors_gap    = 0.05,
    header        = "HR (95% CI)"
  )

fp_render(p3)

favors_gap = 0.05 leaves a small gap around the reference line so the arrowheads do not overlap the dashed ref line. Set favors_gap = FALSE (the default) to start arrows exactly at the reference line.

4 fp_aes — map data columns to visual properties

fp_aes() maps column names in the data frame to aesthetic properties of CI points or text labels. Every value in the named column is applied to the corresponding row.

Supported aesthetics for add_ci(): colour, fill, alpha, glyph, shape, point_size, line_width.

Supported aesthetics for add_text(): colour, fontface, size.

df_aes <- df
df_aes$ci_colour <- ifelse(df$pval < 0.05, "#c0392b", "#2c3e50")
df_aes$ci_fill   <- ifelse(df$pval < 0.05, "#e74c3c", "#7f8c8d")
df_aes$lbl_face  <- ifelse(df$subgroup == "Overall", "bold", "plain")

p4 <- forest_plot(df_aes) |>
  add_text("subgroup",
    header  = "Subgroup",
    width   = 2,
    mapping = fp_aes(fontface = "lbl_face")
  ) |>
  add_ci("HR", "LCI", "UCI",
    trans    = "log",
    xlim     = c(0.2, 3),
    breaks   = c(0.25, 0.5, 1, 2),
    ref_line = 1,
    show_axis = TRUE,
    header   = "HR (95% CI)",
    mapping  = fp_aes(colour = "ci_colour", fill = "ci_fill")
  )

fp_render(p4)

Rows with p < 0.05 are highlighted in red; the “Overall” row is bold. The fp_aes() object simply stores column names as strings — the builder resolves each name at render time.

5 add_header_group — spanning header rows

add_header_group() places a label that spans multiple adjacent panels in a row above the regular column headers. Panels are identified by their 1-based position in the chain of add_*() calls.

df2 <- df
df2$HR2  <- df$HR  * runif(7, 0.85, 1.15)
df2$LCI2 <- df$LCI * runif(7, 0.85, 1.15)
df2$UCI2 <- df$UCI * runif(7, 0.85, 1.15)

set.seed(42)
df2$HR2  <- df$HR  * runif(7, 0.85, 1.15)
df2$LCI2 <- pmax(df2$HR2 - (df$UCI - df$HR) * runif(7, 0.9, 1.1), 0.1)
df2$UCI2 <- df2$HR2 + (df$UCI - df$HR) * runif(7, 0.9, 1.1)

p5 <- forest_plot(df2) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>             # panel 1
  add_gap(width = 0.1) |>                                              # panel 2
  add_ci("HR",  "LCI",  "UCI",                                        # panel 3
    trans = "log", xlim = c(0.2, 3), breaks = c(0.5, 1, 2),
    ref_line = 1, show_axis = TRUE, header = "Endpoint A") |>
  add_ci("HR2", "LCI2", "UCI2",                                       # panel 4
    trans = "log", xlim = c(0.2, 3), breaks = c(0.5, 1, 2),
    ref_line = 1, show_axis = TRUE, header = "Endpoint B") |>
  add_header_group("Primary Endpoints", panels = 3:4,
    background = "#f0f4ff", border = TRUE)

fp_render(p5)

panels = 3:4 spans the two CI panels. The background fill and border line visually group the columns. panels must be contiguous integers.

6 Custom panels with fp_custom

fp_custom() accepts an arbitrary plot_fn that receives the data frame and theme, then returns a ggplot. The y-axis must span 1 to n_rows using the same coordinate space as the built-in panels (row 1 is the top row in the coordinate system after layout, i.e., rev(seq_len(n_rows))).

pval_panel <- fp_custom(
  plot_fn = function(data, n_rows, theme) {
    df_p <- data.frame(
      x     = 0.5,
      y     = rev(seq_len(n_rows)),
      label = ifelse(
        is.na(data$pval), "",
        ifelse(data$pval < 0.001, "< 0.001",
               sprintf("%.3f", data$pval))
      )
    )
    ggplot(df_p, aes(x = x, y = y, label = label)) +
      geom_text(
        size   = theme$text_size,
        colour = theme$text_colour,
        family = theme$base_family
      ) +
      scale_x_continuous(limits = c(0, 1), expand = expansion(0)) +
      theme_void()
  },
  header = "P-value",
  width  = 1.2
)

p6 <- forest_plot(df) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>
  add_ci("HR", "LCI", "UCI",
    trans = "log", xlim = c(0.2, 3), breaks = c(0.25, 0.5, 1, 2),
    ref_line = 1, show_axis = TRUE, header = "HR (95% CI)") |>
  add_custom(pval_panel)

fp_render(p6)

The plot_fn signature is flexible: declare only the parameters you need (data, n_rows, row_heights, theme, spec). panelforest passes each by name, so order does not matter. The function must return a ggplot.

7 Formatter helpers

Three built-in formatters transform numeric columns to display strings without storing pre-formatted text in the data frame.

p7 <- forest_plot(df) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>
  add_text("n",
    header    = "N",
    width     = 0.8,
    align     = "right",
    formatter = fp_fmt_number(digits = 0, big_mark = ",")
  ) |>
  add_text("resp_rate",
    header    = "Response",
    width     = 1.1,
    align     = "right",
    formatter = fp_fmt_percent(digits = 1)
  ) |>
  add_text("pval",
    header    = "P-value",
    width     = 1.2,
    align     = "right",
    formatter = fp_fmt_pvalue(digits = 3, threshold = 0.001)
  ) |>
  add_ci("HR", "LCI", "UCI",
    trans = "log", xlim = c(0.2, 3), breaks = c(0.25, 0.5, 1, 2),
    ref_line = 1, show_axis = TRUE, header = "HR (95% CI)")

fp_render(p7)

Formatter Formats Key arguments
fp_fmt_number() General numeric digits, big_mark, prefix, suffix
fp_fmt_percent() Proportion → percentage digits, scale (default 100), suffix
fp_fmt_pvalue() P-values with threshold digits, threshold, prefix

NA values are always replaced with the na argument (default "").

8 fp_register — extending the rendering pipeline

panelforest dispatches rendering through a type registry. Each spec type ("text", "ci", "dot", etc.) maps to a builder function. fp_register() lets you override a built-in builder or introduce an entirely new type.

The most common use-case is wrapping a built-in builder with pre/post processing. Here we override "ci" to always set a minimum line width:

original_ci_builder <- getFromNamespace(".build_ci", "panelforest")

my_ci_builder <- function(ctx, spec, cell_edits) {
  spec$line_width <- max(spec$line_width, 0.8)
  original_ci_builder(ctx, spec, cell_edits)
}

fp_register("ci", my_ci_builder, overwrite = TRUE)

# Restore the default when done
fp_register("ci", original_ci_builder, overwrite = TRUE)

The same mechanism supports entirely new spec types: create a spec constructor that calls .make_spec() (from the package internals) with a unique type string, then register a builder for that string. Power users can pair fp_custom() with a wrapper spec for full control without touching the registry at all.

9 Dot panel

add_dot() is designed for continuous endpoints such as mean differences or standardised mean differences. It draws a point (with optional CI whiskers) on a linear or log scale.

p9 <- forest_plot(df) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>
  add_ci("HR", "LCI", "UCI",
    trans = "log", xlim = c(0.2, 3), breaks = c(0.25, 0.5, 1, 2),
    ref_line = 1, show_axis = TRUE, header = "HR (95% CI)") |>
  add_gap(width = 0.1) |>
  add_dot("mean_diff",
    lower      = "lci_diff",
    upper      = "uci_diff",
    header     = "Mean Difference",
    ref_line   = 0,
    trans      = "identity",
    colour     = "#2563eb",
    shape      = 21,
    fill       = "#bfdbfe",
    point_size = 2.5,
    breaks     = c(-0.75, -0.5, -0.25, 0, 0.25)
  )

fp_render(p9)

add_dot() and add_ci() share the ref_line, trans, truncate, breaks, colour, fill, shape, and point_size parameters. The key difference is that add_dot() does not render glyphs (no diamond summary style) and does not support show_axis or favors_* annotations.

10 Spacer vs gap

Two spec types create visual white space between panels:

  • add_gap(width) — proportional width, measured in the same relative units as the width arguments of other panels (e.g., width = 0.15 is 15% of a “standard” panel width of 1.0).
  • add_spacer(width, unit) — absolute width in any grid::unit() unit: "mm", "cm", "in", or "pt".
p10 <- forest_plot(df) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>
  add_gap(width = 0.15) |>
  add_text("n", header = "N", width = 0.7, align = "right",
           formatter = fp_fmt_number(digits = 0)) |>
  add_spacer(width = 5, unit = "mm") |>
  add_ci("HR", "LCI", "UCI",
    trans = "log", xlim = c(0.2, 3), breaks = c(0.25, 0.5, 1, 2),
    ref_line = 1, show_axis = TRUE, header = "HR (95% CI)")

fp_render(p10)

Use add_gap() when you want space that scales with the plot width. Use add_spacer() when you need a precise, device-independent margin — for example to align a forest plot with a companion table produced at a fixed column width.

11 Multiple CI panels side by side

Two add_ci() calls produce two independent forest plot panels, each with its own scale, reference line, and axis. This is useful for comparing two endpoints or two subgroup analyses in a single figure.

set.seed(1)
df3 <- df
df3$HR_pfs  <- df$HR * runif(7, 0.80, 1.20)
df3$LCI_pfs <- pmax(df3$HR_pfs - (df$UCI - df$LCI) / 2 * runif(7, 0.9, 1.1), 0.1)
df3$UCI_pfs <- df3$HR_pfs + (df$UCI - df$LCI) / 2 * runif(7, 0.9, 1.1)

p11 <- forest_plot(df3) |>
  add_text("subgroup", header = "Subgroup", width = 2) |>
  add_ci("HR", "LCI", "UCI",
    trans     = "log",
    xlim      = c(0.2, 3),
    breaks    = c(0.25, 0.5, 1, 2),
    ref_line  = 1,
    show_axis = TRUE,
    axis_label = "OS  HR",
    header    = "Overall Survival",
    colour    = "#1d4ed8"
  ) |>
  add_gap(width = 0.15) |>
  add_ci("HR_pfs", "LCI_pfs", "UCI_pfs",
    trans     = "log",
    xlim      = c(0.2, 3),
    breaks    = c(0.25, 0.5, 1, 2),
    ref_line  = 1,
    show_axis = TRUE,
    axis_label = "PFS  HR",
    header    = "Progression-Free Survival",
    colour    = "#15803d"
  ) |>
  add_header_group("Time-to-Event Endpoints", panels = 2:4,
    background = "#f8fafc", border = TRUE)

fp_render(p11)

Each CI panel is fully independent: different colours, separate axes, and separate header labels. A add_header_group() spans both CI panels (and the gap between them) to create a grouped header tier.

Putting it all together

The following example combines multiple advanced features — fp_aes(), add_header_group(), a custom panel, formatters, and favors annotations — into a single publication-style figure.

df_full <- df
df_full$ci_colour <- ifelse(df$pval < 0.05, "#991b1b", "#374151")
df_full$lbl_face  <- ifelse(df$subgroup == "Overall", "bold", "plain")

sig_panel <- fp_custom(
  plot_fn = function(data, n_rows, theme) {
    df_s <- data.frame(
      x     = 0.5,
      y     = rev(seq_len(n_rows)),
      label = ifelse(data$pval < 0.05, "*", "")
    )
    ggplot(df_s, aes(x = x, y = y, label = label)) +
      geom_text(size = theme$text_size * 1.4, colour = "#c0392b",
                family = theme$base_family) +
      scale_x_continuous(limits = c(0, 1), expand = expansion(0)) +
      theme_void()
  },
  header = "",
  width  = 0.4
)

p_full <- forest_plot(df_full) |>
  add_text("subgroup",
    header  = "Subgroup",
    width   = 2,
    mapping = fp_aes(fontface = "lbl_face")
  ) |>
  add_text("n",
    header    = "N",
    width     = 0.7,
    align     = "right",
    formatter = fp_fmt_number(digits = 0, big_mark = ",")
  ) |>
  add_text("resp_rate",
    header    = "Response",
    width     = 1.1,
    align     = "right",
    formatter = fp_fmt_percent(digits = 1)
  ) |>
  add_gap(width = 0.12) |>
  add_ci("HR", "LCI", "UCI",
    trans         = "log",
    xlim          = c(0.2, 3),
    breaks        = c(0.25, 0.5, 1, 2),
    ref_line      = 1,
    show_axis     = TRUE,
    axis_label    = "Hazard Ratio (log scale)",
    favors_left   = "Favors Treatment",
    favors_right  = "Favors Control",
    favors_span   = 0.60,
    favors_gap    = 0.06,
    header        = "HR (95% CI)",
    mapping       = fp_aes(colour = "ci_colour")
  ) |>
  add_text("pval",
    header    = "P-value",
    width     = 1.2,
    align     = "right",
    formatter = fp_fmt_pvalue(digits = 3, threshold = 0.001)
  ) |>
  add_custom(sig_panel) |>
  add_header_group("Summary Statistics", panels = 2:3,
    background = "#f0fdf4", border = TRUE) |>
  add_header_group("Efficacy", panels = 5:7,
    background = "#eff6ff", border = TRUE)

fp_render(p_full)

This figure demonstrates:

  • Per-row colour via fp_aes() (red for significant subgroups)
  • Bold formatting for the overall row via fp_aes(fontface = ...)
  • fp_fmt_number() and fp_fmt_percent() for compact display
  • fp_fmt_pvalue() with a < 0.001 threshold
  • A custom significance-star panel via fp_custom()
  • Two spanning header groups via add_header_group()
  • favors_left / favors_right annotations with a gap
